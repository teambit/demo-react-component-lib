---
description: Opens and closes a container box
---

import { Containee } from '@teambit/base-ui.surfaces.abs-container';
import { useState } from 'react';
import { Drawer } from './drawer';

#### Using the drawer component

```jsx live
// import { Containee } from '@teambit/base-ui.surfaces.abs-container';

() => {
  const frameStyle = { border: '1px solid gray', borderRadius: 5, padding: 2 };
  const [isOpen, setIsOpen] = useState(true);

  return (
    <div style={{ padding: 30 }}>
      <Drawer
        open={isOpen}
        onChange={(event, open) => setIsOpen(open)}
        placeholder="pre-opened"
        style={frameStyle}
      >
        <Containee style={frameStyle}>menu</Containee>
      </Drawer>
    </div>
  );
};
```

#### Using the drawer component with 'click outside' disabled

Set `clickOutside` to false to disable the removal of `Containee` by click outside of it.

```jsx live
// import { Containee } from '@teambit/base-ui.surfaces.abs-container';

() => {
  const [isOpen, setIsOpen] = useState(true);

  return (
    <div style={{ padding: 30 }}>
      <Drawer
        open={isOpen}
        onChange={(event, open) => setIsOpen(open)}
        clickOutside={false}
        placeholder="no click outside"
        style={{ border: '1px solid gray', borderRadius: 5, padding: 4 }}
      >
        <Containee>menu</Containee>
      </Drawer>
    </div>
  );
};
```
