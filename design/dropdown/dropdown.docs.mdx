import { Theme } from '@learn-harmony/design.styles.theme';
import { Dropdown } from './dropdown';
import { Item } from './item';
import { dropdownList } from './dropdown.mock';
import { hoverable } from '@learn-harmony/design.styles.characteristics.hoverable';

### Usage example for the Dropdown

```ts live
<Theme>
  <Dropdown
    placeholder={false}
    placeholderTitle={
      <Item image="https://static.bit.dev/bit-logo.svg" name="teambit" />
    }
  >
    {dropdownList.map((listItem, index) => (
      <Item
        key={index}
        image={listItem.image}
        name={listItem.name}
        className={hoverable}
      />
    ))}
  </Dropdown>
</Theme>
```

### Usage example for the Item

The dropdown also provides an `Item` component, which is separately available for usage:

```ts live
<Theme>
  <Item
    className={hoverable}
    image="https://static.bit.dev/bit-logo.svg"
    name="teambit"
  />
</Theme>
```
