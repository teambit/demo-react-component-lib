import { Dropdown } from './dropdown';
import { Item } from './item';
import { dropdownList } from './dropdown.mock';
import { hoverable } from '@learn-harmony/base-ui.styles.characteristics.hoverable';

### Usage example for the Dropdown

```ts live
<Dropdown
  placeholder={false}
  placeholderTitle={
    <Item image="https://static.bit.dev/bit-logo.svg" name="teambit" />
  }
>
  {dropdownList.map((listItem, index) => (
    <Item
      key={index}
      image={listItem.image}
      name={listItem.name}
      className={hoverable}
    />
  ))}
</Dropdown>
```

### Usage example for the Item

The dropdown also provides an `Item` component, which is separetaly avilable for usage:

```ts live
<Item
  className={hoverable}
  image="https://static.bit.dev/bit-logo.svg"
  name="teambit"
/>
```
